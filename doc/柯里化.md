# 柯里化

## 柯里化

```js
function curryFn(fn) {
  return function curriedFn(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args)
    } else {
      return function (...args1) {
        return curriedFn.apply(this, [...args, ...args1])
      }
    }
  }
}
// 测试
function add(a, b) {
  return a + b
}
const addCurried = curryFn(add)
console.log(addCurried(1, 2))
console.log(addCurried(1)(2))
```
